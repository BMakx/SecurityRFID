/*
 * Autoryzacja1.c
 * MASTER
 *
 * PINOUT SETUP
 * ----------------------------
 * LCD - PORT C
 * P7    P6 P5 P4 P3 P2 P1 P0
 * RW(0) RS E  L  D7 D6 D5 D4
 * ----------------------------
 * LED - PORT A (PA0-PA7) - Active Low (0 zapala diodę)
 */

#include <avr/io.h>
#define F_CPU 11059200UL
#include <util/delay.h>
#include <stdint.h>
#include <string.h>

#define BAUD 19200UL
#define UBRR_VALUE ((F_CPU / 16UL / BAUD) - 1)
#define SLAVE_ADDR "00126f9e3e5c"
#define ID_DO_WYSLANIA "OCHRONIARZ 67"

// --- Deklaracje funkcji ---
void LCD_init(void);
void LCD_send_4bits(uint8_t bits, uint8_t rs);
void LCD_send_byte(uint8_t byte, uint8_t rs);
void LCD_write_string(const char* s);
void LCD_command(uint8_t command);
void LCD_clear(void);
void LCD_set_cursor(uint8_t row, uint8_t col);

void UART_init(void);
void UART_send_char(char data);
void UART_send_string(const char *str);
uint8_t UART_char_available(void);
void UART_flush(void);
void UART_read_line(char* buffer, uint8_t max_len, uint16_t timeout_ms);

void BT_send_command(const char *cmd);
uint8_t BT_wait_keyword(const char *keyword, uint16_t timeout_ms);
void BTM222_init_master(void);
void BTM222_connect(const char *addr);
uint8_t BTM222_handshake(void);

void Signal_OK(void);
void Signal_REJECT(void);

// --- Funkcje LED ---
void Signal_REJECT(void) {
    // Efekt fali przez ok. 3 sekundy
    // 8 diod * 100ms = 800ms na jeden pełny przebieg. 4 przebiegi = 3.2s
    for(uint8_t cycle = 0; cycle < 4; cycle++) {
        for(uint8_t i = 0; i < 8; i++) {
            PORTA = ~(1 << i); // Zapal tylko i-tą diodę (negacja bitu)
            _delay_ms(100);
        }
    }
    PORTA = 0xFF; // Zgaś wszystko na koniec
}

void Signal_OK(void) {
    // 3 sekundy mrugania: 6 cykli po 500ms (250ms ON + 250ms OFF)
    for(uint8_t i = 0; i < 6; i++) {
        PORTA = 0x00; // Wszystkie ON (logiczne 0)
        _delay_ms(250);
        PORTA = 0xFF; // Wszystkie OFF (logiczne 1)
        _delay_ms(250);
    }
}

// --- MAIN ---
int main(void) {
    // Konfiguracja LED
    DDRA = 0xFF;  // Port A jako wyjście
    PORTA = 0xFF; // Wszystkie diody wyłączone (Active Low)

    LCD_init();
    UART_init();

    LCD_clear();
    LCD_set_cursor(0, 0);
    LCD_write_string("MASTER START");

    BTM222_init_master();
    
    LCD_clear();
    LCD_write_string("Laczenie...");
    BTM222_connect(SLAVE_ADDR);

    if (BTM222_handshake()) {
        
        // --- GŁÓWNA PĘTLA POWTARZALNOŚCI ---
        while (1) {
            LCD_clear();
            LCD_set_cursor(0, 0);
            LCD_write_string("Wysylam ID...");
            
            _delay_ms(500); 
            UART_send_string(ID_DO_WYSLANIA);
            UART_send_string("\r\n");

            LCD_clear();
            LCD_write_string("ID: OCHRONIARZ");
            LCD_set_cursor(1, 0);
            LCD_write_string("Czekam na decyzje");

            char odpowiedz[16];
            UART_read_line(odpowiedz, sizeof(odpowiedz), 0); // Czeka na odpowiedz

            LCD_clear();
            if (strcmp(odpowiedz, "OK") == 0) {
                LCD_write_string("DOSTEP PRZYZNANY");
                Signal_OK();
            } 
            else if (strcmp(odpowiedz, "ODRZUCONO") == 0) {
                LCD_write_string("BRAK DOSTEPU");
                Signal_REJECT();
            }

            LCD_set_cursor(1, 0);
            LCD_write_string("Restart za 5s...");
            
            // Wymagane 5 sekund przerwy przed ponownym wysłaniem
            _delay_ms(5000); 
        }
    } else {
        LCD_clear();
        LCD_write_string("BLAD POLACZENIA");
    }

    while (1) { } // Fallback
    return 0;
}

// --- Implementacje ---
void UART_init(void) {
    UBRRH = (unsigned char)(UBRR_VALUE >> 8);
    UBRRL = (unsigned char)UBRR_VALUE;
    UCSRB = (1 << RXEN) | (1 << TXEN);
    UCSRC = (1 << URSEL) | (1 << UCSZ0) | (1 << UCSZ1);
}
void UART_send_char(char data) {
    while (!(UCSRA & (1 << UDRE))) { }
    UDR = data;
}
void UART_send_string(const char *str) {
    while (*str) UART_send_char(*str++);
}
uint8_t UART_char_available(void) {
    return (UCSRA & (1 << RXC)) ? 1 : 0;
}
void UART_flush(void) {
    while (UART_char_available()) { (void)UDR; }
}
void UART_read_line(char* buffer, uint8_t max_len, uint16_t timeout_ms) {
    uint8_t i = 0; buffer[0] = '\0';
    while (timeout_ms > 0 || timeout_ms == 0) {
        if (UART_char_available()) {
            char c = UDR;
            if (c == '\r' || c == '\n') { 
                if (i > 0) { buffer[i] = '\0'; return; } 
            } else if (i < max_len - 1) { 
                buffer[i++] = c; 
            }
        } else { 
            _delay_ms(1); 
            if (timeout_ms > 0) timeout_ms--; 
        }
    }
}
void BT_send_command(const char *cmd) { 
    UART_send_string(cmd); UART_send_char('\r'); 
}
uint8_t BT_wait_keyword(const char *keyword, uint16_t timeout_ms) {
    uint8_t idx = 0; uint8_t key_len = (uint8_t)strlen(keyword); uint16_t t = timeout_ms;
    while (t > 0) {
        if (UART_char_available()) {
            char c = UDR;
            if (c == keyword[idx]) { idx++; if (idx == key_len) return 1; }
            else { idx = (c == keyword[0]) ? 1 : 0; }
        } else { _delay_ms(1); t--; }
    } return 0;
}
void BTM222_init_master(void) {
    _delay_ms(1000); UART_flush();
    BT_send_command("AT"); _delay_ms(200);
    BT_send_command("ATE0"); _delay_ms(200);
    BT_send_command("ATQ0"); _delay_ms(500);
    BT_send_command("ATN=MASTER"); _delay_ms(500);
    BT_send_command("ATR=0"); _delay_ms(500); // ATR=0 / ATR0 (zgodnie z poprzednimi ustaleniami)
    BT_send_command("ATP=1234"); _delay_ms(500);
    BT_send_command("ATZ"); _delay_ms(3000); UART_flush();
}
void BTM222_connect(const char *addr) {
    UART_flush();
    UART_send_string("ATD");
    UART_send_string(addr); // Pamiętaj: ATD + adres (bez =) jeśli taka była działająca składnia z Twoich testów
    UART_send_char('\r');
    _delay_ms(8000); 
}
uint8_t BTM222_handshake(void) {
    uint8_t tries = 8;
    while (tries--) {
        UART_send_string("PING\r\n");
        if (BT_wait_keyword("READY", 1500)) return 1;
    } return 0;
}
void LCD_init() {
    DDRC = 255; PORTC = 0; _delay_ms(40);
    PORTC = 0b00100011; PORTC = 0b00000011; _delay_ms(5);
    PORTC = 0b00100011; PORTC = 0b00000011; _delay_us(100);
    PORTC = 0b00100011; PORTC = 0b00000011; _delay_us(50);
    PORTC = 0b00100010; PORTC = 0b00000010;
    PORTC = 0b00101100; PORTC = 0b00001100; _delay_ms(50);
    PORTC = 0b00100000; PORTC = 0b00000000;
    PORTC = 0b00101100; PORTC = 0b00001100; _delay_ms(50);
    PORTC = 0b00100000; PORTC = 0b00000000;
    PORTC = 0b00100001; PORTC = 0b00000001; _delay_ms(100);
    PORTC = 0b00100001; PORTC = 0b00000001;
    PORTC = 0b00101000; PORTC = 0b00001000; _delay_ms(50);
    PORTC = 0b01100000; PORTC = 0b01000100;
    PORTC = 0b01100001; PORTC = 0b01000001; _delay_ms(50);
}
void LCD_command(uint8_t command) { LCD_send_byte(command, 0); _delay_ms(2); }
void LCD_send_4bits(uint8_t bits, uint8_t rs) {
    uint8_t port = 0; if (rs) port |= (1 << 6); port |= (1 << 5); port |= (bits >> 4);
    PORTC = port; _delay_ms(1); PORTC &= ~(1 << 5);
}
void LCD_send_byte(uint8_t byte, uint8_t rs) { LCD_send_4bits(byte & 0xF0, rs); LCD_send_4bits(byte << 4, rs); }
void LCD_write_string(const char* s) { while (*s) LCD_send_byte(*s++, 1); }
void LCD_clear() { LCD_command(0x01); _delay_ms(2); }
void LCD_set_cursor(uint8_t row, uint8_t col) { uint8_t address = (row == 0 ? 0x80 : 0xC0) + col; LCD_command(address); }
